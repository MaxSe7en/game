.tab-list { list-style-type: none; margin: 0; padding: 0; display: flex; } .tab-list li { margin-right: 20px; padding: 10px 20px; background-color: #eee; cursor: pointer; } .tab-list li.active { background-color: #ddd; } .tab-panel { display: none; } .tab-panel.active { display: block; } .tab-menu { list-style-type: none; margin: 0; padding: 0; overflow: hidden; background-color: #f1f1f1; } .tab-menu li { float: left; } .tab-menu li a { display: block; color: #000; text-align: center; padding: 14px 16px; text-decoration: none; border-bottom: 2px solid transparent; } .tab-menu li.active a { border-bottom: 2px solid #000; } .tab-content { display: block; padding: 20px; } .active-tab { display: block; } // Get the list of tabs const tabs = document.querySelectorAll(".tab-menu li"); // Get the content panels const panels = document.querySelectorAll(".tab-content .tab-panel"); // Add click event listener to each tab tabs.forEach((tab, index) => { tab.addEventListener("click", () => { // Remove active class from all tabs and panels tabs.forEach((tab) => tab.classList.remove("active")); panels.forEach((panel) => panel.classList.remove("active-tab")); // Add active class to clicked tab and panel tab.classList.add("active"); panels[index].classList.add("active-tab"); }); });
